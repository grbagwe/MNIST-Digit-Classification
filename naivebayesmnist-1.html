# we know that P(xi/Ck) that is the feature given the class follows 
# normal distribution whose mean and variance we have already calculated 
# We get the value of P(C/x) as the Multiplication of N(mui,sigmai | C)
# This is than compared with P(C/x) obtained from the other classes 
# which ever fits sharply is the Class of the Test variablezeros
txlen=Txl.shape[0]                      # number of rows of the Test labels 
probpred = np.zeros((txlen,1))          # initiaise the matrix to store the predictions 
for i in range(txlen):         
    probmat = np.zeros((10,flen))
    probd = np.zeros((10,1))        
    for d in range(10):             # for all classes ie 10 in this case
        for f in range(flen):
            if (Trvar[d,f]!=0):        # Check if variance is zero
        # if the variance is zero just put a small value so that it will not go to infinity         
                 temp1 = (1/np.sqrt(Trvar[d,f]))
        # Since (2 *pi) is constant in al the calculations they can be ommited 

                 temp2 = np.exp(-(pow(Txd[i][f]-Trmean[d,f],2))/(2*Trvar[d,f]))
        # the value on the exp, since we take log to convert multiplication to summation, 
        # exp vanishes as log(exp(a)) =a
                 if (temp2>0):
                    probmat[d,f] = -np.log(temp1) - np.log(temp2)
            else:
                # if Varaince is zero
                probmat[d,f] = -np.log(0.01)
        probd[d] = np.sum(probmat[d,:])
         # by Naive bayes 
    probpred[i] = np.argmin(probd)
      #the min value is the value that matches most with a certain class 
crtcount = 0  # intialize correct count variable
for x in range(Txl.shape[0]):
    if (int(probpred[x])==int(Txl[x])): # if the prediction matches the Test label   
        crtcount =crtcount +1;
